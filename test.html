Select Box Substitution

works in ie7+, firefox10+, chrome20+

initialize one individual select with auto submit

new CI.StyleableSelective.List(document.getElement('#oneList'), {
	autoSubmit: true,
	onOpened: function() {
		this.substitute.box.addClass('open');
	},
	onClosed: function(){
		this.substitute.box.removeClass('open');
	}
});


initialize one individual select with auto submit per ajax

new CI.StyleableSelective.List(document.getElement('#ajax'), {
	onOpened: function() {
		this.substitute.box.addClass('open');
	},
	onClosed: function(){
		this.substitute.box.removeClass('open');
	},
	onSelected: function(){
		this.selectBox.getParent('form').send();
	}
});


initialize a group of individual selects with fixed width

new CI.StyleableSelective.LinkedLists(document.getElements('select.substitute'), {
	variableWidth: false,
	onOpened: function() {
		this.substitute.box.addClass('open');
	},
	onClosed: function(){
		this.substitute.box.removeClass('open');
	}
});




initialize a group of linked selects with variable width

CI.StyleableSelective.Lists(document.getElements('select.linkedSubstitute'), { 
	onOpened: function() {
		this.substitute.box.addClass('open');
	},
	onClosed: function(){
		this.substitute.box.removeClass('open');
	}
});




initialize a group of linked selects with variable width and auto submit per ajax

CI.StyleableSelective.Lists(document.getElements('select.linkedSubstituteAjax'), { 
	onOpened: function() {
		this.substitute.box.addClass('open');
	},
	onClosed: function(){
		this.substitute.box.removeClass('open');
	},
	onSelected: function(){
		this.selectBoxes.pick().getParent('form').send();
	}
});
